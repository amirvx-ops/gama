WITH latest_meetings AS (SELECT DISTINCT ON (record_id) * FROM gama_meetings ORDER BY record_id, created_at DESC) SELECT COUNT(*) AS total_meetings, COUNT(*) FILTER (WHERE meeting_status = 'התקיימה') AS held, COUNT(*) FILTER (WHERE meeting_status = 'לא התקיימה') AS not_held, COUNT(*) FILTER (WHERE meeting_status = 'נדחתה') AS postponed, ROUND(COUNT(*) FILTER (WHERE meeting_status = 'התקיימה') * 100.0 / NULLIF(COUNT(*), 0), 1) AS held_percentage FROM latest_meetings
